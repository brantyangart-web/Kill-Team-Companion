          const filter = audioCtx.createBiquadFilter();
          filter.type = 'bandpass'; filter.frequency.value = 220;
          const noiseGain = audioCtx.createGain();
          noiseGain.gain.setValueAtTime(0.12, now);
          noiseGain.gain.exponentialRampToValueAtTime(0.0001, now + 0.12);
          noise.connect(filter); filter.connect(noiseGain); noiseGain.connect(audioCtx.destination);
          noise.start();
        }
      } catch (e) {}

      applyWounds(amount, manualDrRolls = null) {
        if (this.isDead) return 0;
        
        let actualDamage = 0;
        addLog(`[ä¼¤å®³] ${this.name} å‡†å¤‡åˆ†é… ${amount} ç‚¹ä¼¤å®³...`);

        if (this.faction === gameState.factionB) {
          const hasPloyActive = gameState.pmActivePloys.includes('contagious_resilience');
          addLog(`[ç‰¹æ€§] è§¦å‘ç˜Ÿç–«å®ˆå«ä¸“å±žã€æ¶å¿ƒä½œå‘• (DR 5+)ã€‘ ${hasPloyActive ? '(å·²å¼€å¯ä¼ æŸ“éŸ§æ€§ï¼Œå…è®¸é¦–ä¸ªå¤±è´¥é‡æŠ•)' : ''}ï¼š`);
          
          let drRollIndex = 0;
          let hasRerolled = false;

          for (let i = 0; i < amount; i++) {
            let roll;
            if (manualDrRolls && drRollIndex < manualDrRolls.length) {
              roll = manualDrRolls[drRollIndex++];
              addLog(`  - ä¼¤å®³ç‚¹ #${i+1}: æ‰‹åŠ¨å½•å…¥éª°å­ [${roll}]`);
            } else {
              roll = Math.floor(Math.random() * 6) + 1;
              addLog(`  - ä¼¤å®³ç‚¹ #${i+1}: æŠ•éª° [${roll}]`);
            }

            if (roll < 5 && hasPloyActive && !hasRerolled && !manualDrRolls) {
              hasRerolled = true;
              const oldRoll = roll;
              roll = Math.floor(Math.random() * 6) + 1;
              addLog(`    -> ðŸ›¡ï¸ è§¦å‘ã€ä¼ æŸ“éŸ§æ€§ã€‘ï¼è‡ªåŠ¨é‡æŠ•å¤±è´¥éª° [${oldRoll}] -> [${roll}]`);
            }

            if (roll >= 5) {
              addLog(`    -> å…é™¤è¯¥ç‚¹ä¼¤å®³ï¼`);
              playSound('bubble');
            } else {
              addLog(`    -> å‡å…å¤±è´¥ã€‚`);
              actualDamage++;
              playSound('flesh');
            }
          }
        } else {
          actualDamage = amount;
          if (actualDamage > 0) playSound('flesh');
        }

        const prevHp = this.wounds;
        this.wounds = Math.max(0, this.wounds - actualDamage);
        addLog(`[åˆ†é…] ${this.name} ç”Ÿå‘½å€¼: ${prevHp} -> ${this.wounds} ${this.wounds === 0 ? '(å·²é˜µäº¡!)' : ''}`);
        
        if (this.wounds === 0) {
          this.isDead = true;
          this.hasActed = true;
          triggerOperativeDeathOverlay(this);
          setTimeout(() => checkGameOver(), 500);
        }
        return actualDamage;
      }
    }

    function endTurningPoint() {
      playSound('click');
      if (gameState.turningPoint >= 4) {
        endGame('TP5');
        return;
      }
      gameState.turningPoint += 1;
      gameState.phase = 'Initiative';
      
      // CP gains at start of TP
      gameState.smCp += 1;
      gameState.pmCp += 1;
      
      // Reset active ploys
      gameState.smActivePloys = [];
      gameState.pmActivePloys = [];

      // Reset operatives
      gameState.operatives.forEach(op => {
        if (!op.isDead) {
          op.hasActed = false;
          op.apl = op.maxApl;
          op.actionsPerformed = [];
        }
      });

      const nextBtn = document.getElementById('btn-next-phase');
      if (nextBtn) nextBtn.style.display = 'none';

      addLog(`\n========================================`);
      addLog(`>>> Turning Point ${gameState.turningPoint} å¼€å§‹ï¼`);
      addLog(`>>> åŒæ–¹å„èŽ·å¾— 1 CPã€‚`);
      addLog(`========================================`);

    }

    function checkGameOver() {
      if (gameState.gameOver) return;
      const factions = [gameState.factionA, gameState.factionB];
      const aliveCounts = {};
      factions.forEach(f => {
        aliveCounts[f] = gameState.operatives.filter(o => o.faction === f && !o.isDead).length;
      });

      if (aliveCounts[gameState.factionA] === 0 && aliveCounts[gameState.factionB] === 0) {
        endGame('DRAW_WIPEOUT');
      } else if (aliveCounts[gameState.factionA] === 0) {
        endGame('PM_WIPEOUT');
      } else if (aliveCounts[gameState.factionB] === 0) {
        endGame('SM_WIPEOUT');
      }
    }

    function calcAttackStats() {
      wizardState.attackCrit = 0;
      wizardState.attackNorm = 0;
      wizardState.attackRolls.forEach(val => {
        if (val === 6) wizardState.attackCrit++;
        else if (val >= wizardState.weapon.ts) wizardState.attackNorm++;
      });
    }

    function calcDefenseStats() {
      wizardState.defCrit = 0;
      wizardState.defNorm = 0;
      wizardState.defenseRolls.forEach(val => {
        if (val === 6) wizardState.defCrit++;
        else if (val >= wizardState.defender.sv) wizardState.defNorm++;
      });
    }

    function calcMeleeStats() {
      wizardState.attackCrit = 0;
      wizardState.attackNorm = 0;
      wizardState.attackRolls.forEach(val => {
        if (val === 6) wizardState.attackCrit++;
        else if (val >= wizardState.weapon.ts) wizardState.attackNorm++;
      });
      
      wizardState.defCrit = 0;
      wizardState.defNorm = 0;
      const defMeleeWeapon = wizardState.defender.weapons.filter(w => !w.isRanged)[0] || new Weapon('é‡æ‹³ (Fists)', 3, 3, 3, 4, false);
      wizardState.defenseRolls.forEach(val => {
        if (val === 6) wizardState.defCrit++;
        else if (val >= defMeleeWeapon.ts) wizardState.defNorm++;
      });
    }

    function inputPhysicalDice(type, val) {
      playSound('click');
      if (type === 'attack') {
        if (wizardState.attackRolls.length < wizardState.weapon.attacks) {
          wizardState.attackRolls.push(val);
          calcAttackStats();
          if (wizardState.attackRolls.length === wizardState.weapon.attacks) playSound('crit');
          renderShootStep();
        }
      } else if (type === 'defense') {
        let dfCount = wizardState.defender.df;
        if (wizardState.inCover) dfCount = Math.max(0, dfCount - 1);
        if (wizardState.defenseRolls.length < dfCount) {
          wizardState.defenseRolls.push(val);
          calcDefenseStats();
          if (wizardState.defenseRolls.length === dfCount) playSound('crit');
          renderShootStep();
        }
      } else if (type === 'meleeAttack') {
        if (wizardState.attackRolls.length < wizardState.weapon.attacks) {
          wizardState.attackRolls.push(val);
          calcMeleeStats();
          renderFightStep();
        }
      } else if (type === 'meleeDefense') {
        const defMeleeWeapon = wizardState.defender.weapons.filter(w => !w.isRanged)[0] || new Weapon('é‡æ‹³ (Fists)', 3, 3, 3, 4, false);
        if (wizardState.defenseRolls.length < defMeleeWeapon.attacks) {
          wizardState.defenseRolls.push(val);
          calcMeleeStats();
          renderFightStep();
        }
      }
    }

    function clearPhysicalDice(type) {
      playSound('click');
      if (type === 'attack') { wizardState.attackRolls = []; calcAttackStats(); renderShootStep(); }
      else if (type === 'defense') { wizardState.defenseRolls = []; calcDefenseStats(); renderShootStep(); }
      else if (type === 'meleeAttack') { wizardState.attackRolls = []; calcMeleeStats(); renderFightStep(); }
      else if (type === 'meleeDefense') { wizardState.defenseRolls = []; calcMeleeStats(); renderFightStep(); }
    }

    function rollAttackDice() {
      const nextBtn = document.getElementById('modal-btn-next');
      const pool = document.getElementById('attack-dice-pool');
      const rollBtn = document.getElementById('btn-roll-attack');

      rollBtn.disabled = true;
      nextBtn.disabled = true;

      const attDiceClass = wizardState.attacker.faction === gameState.factionA ? 'sm-dice' : 'pm-dice';

      // 1. åˆå§‹åŒ–æ»šåŠ¨çš„å ä½éª°å­
      pool.innerHTML = '';
      const totalAttacks = wizardState.weapon.attacks;
      for (let i = 0; i < totalAttacks; i++) {
        pool.innerHTML += `<div class="kt-dice-cube ${attDiceClass} rolling">?</div>`;
      }

      // 2. æ»šåŠ¨èµ·æ‰‹ç‰¹æ•ˆä¸Žå£°éŸ³
      triggerCombatVisual("ðŸ”¥ OPEN FIRE!", "shoot");
      playSound('shoot');

      // 3. é¡ºåºåœä¸‹ (é€ä¸ªç¡®å®šç‚¹æ•°å¹¶æ’­æ”¾éŸ³æ•ˆ)
      const finalRolls = [];
      let currentSettleIndex = 0;

      function settleNextDice() {
        if (currentSettleIndex < totalAttacks) {
          const val = Math.floor(Math.random() * 6) + 1;
          finalRolls.push(val);
          
          // æ›´æ–° DOM
          const diceCubes = pool.getElementsByClassName('kt-dice-cube');
          const cube = diceCubes[currentSettleIndex];
          cube.classList.remove('rolling');
          cube.textContent = val;
          
          if (val === 6) {
            cube.classList.add('crit-dice');
            playSound('crit');
          } else if (val < wizardState.weapon.ts) {
            cube.classList.add('fail-dice');
            playSound('click');
          } else {
            playSound('click');
          }

          currentSettleIndex++;
          // 400ms é¡ºåºå»¶è¿Ÿ
          setTimeout(settleNextDice, 400);
        } else {
          // å…¨éƒ¨æŽ·éª°å®Œæ¯•ï¼Œè®¡ç®—å¹¶å‘ˆçŽ°ç»“æžœ
          wizardState.attackRolls = finalRolls;
          recalculateAttackStats();
          
          const successes = wizardState.attackCrit + wizardState.attackNorm;
          if (successes === 0) {
            playSound('epic_fail');
            triggerCombatVisual("âŒ ALL MISSED!", "normal");

    function parseManualAttack() {
      const inputEl = document.getElementById('manual-att-dice-val');
      if (!inputEl) return;
      const valStr = inputEl.value;
      const rolls = valStr.split(',').map(s => parseInt(s.trim(), 10)).filter(n => !isNaN(n) && n >= 1 && n <= 6);
      wizardState.attackRolls = rolls;
      recalculateAttackStats();
    }

    function parseManualDefense() {
      const inputEl = document.getElementById('manual-def-dice-val');
      if (!inputEl) return;
      const valStr = inputEl.value;
      const rolls = valStr.split(',').map(s => parseInt(s.trim(), 10)).filter(n => !isNaN(n) && n >= 1 && n <= 6);
      wizardState.defenseRolls = rolls;
      recalculateDefenseStats();
    }

    function recalculateAttackStats() {
      let crits = 0;
      let norms = 0;
      const ts = wizardState.weapon.ts;
      wizardState.attackRolls.forEach(val => {
        if (val === 6) crits++;
        else if (val >= ts) norms++;
      });
      wizardState.attackCrit = crits;
      wizardState.attackNorm = norms;
    }

    function recalculateDefenseStats() {
      let crits = 0;
      let norms = wizardState.inCover ? 1 : 0; // Cover provides 1 free normal save
      const sv = wizardState.defender.sv;
      wizardState.defenseRolls.forEach(val => {
        if (val === 6) crits++;
        else if (val >= sv) norms++;
      });
      wizardState.defCrit = crits;
      wizardState.defNorm = norms;
    }

    function getDuelAvatarHtml(opId, faction) {
      const avatarUrl = gameState.customAvatars[opId];
      let fallbackUrl = faction === gameState.factionA ? './default_sm_avatar.png' : './default_pm_avatar.png';
      
      const activeOp = gameState.operatives.find(o => o.id === opId);
      if (activeOp && activeOp.defaultAvatar) {
        fallbackUrl = activeOp.defaultAvatar;
      } else {
        const template = SM_TEMPLATES.concat(PM_TEMPLATES).find(t => t.id === opId);
        if (template && template.defaultAvatar) {

                <img src="${imgUrl}" class="op-avatar-img" />
              </div>`;
    }

    function chooseMeleeDice(side, idx) {
      if (side !== wizardState.meleeTurn) {
        playSound('alert');
        alert('çŽ°åœ¨ä¸å±žäºŽä½ çš„è¿‘æˆ˜åˆ†é…å›žåˆï¼');
        return;
      }

      const diceList = side === 'attacker' ? wizardState.activeAttackerDice : wizardState.activeDefenderDice;

    function resolveMeleeChoice(action) {
      if (!wizardState.selectedMeleeDice) return;
      const { side, idx } = wizardState.selectedMeleeDice;
      
      const diceList = side === 'attacker' ? wizardState.activeAttackerDice : wizardState.activeDefenderDice;
      const dice = diceList[idx];
      if (dice.used) return;

      const targetOpponent = side === 'attacker' ? wizardState.defender : wizardState.attacker;
      const opponentDiceList = side === 'attacker' ? wizardState.activeDefenderDice : wizardState.activeAttackerDice;
      
      let activeWeapon;
      if (side === 'attacker') {
        activeWeapon = wizardState.weapon;
      } else {
        activeWeapon = wizardState.defender.weapons.filter(w => !w.isRanged)[0] || { normalDamage: 3, criticalDamage: 4 };
      }

      if (!wizardState.meleeLogs) wizardState.meleeLogs = '';

      if (action === 'strike') {
        dice.used = true;
        const dmg = dice.isCrit ? activeWeapon.criticalDamage : activeWeapon.normalDamage;
        const msg = `> ${side === 'attacker' ? 'æ”»å‡»æ–¹' : 'é˜²å®ˆæ–¹'} æ‰§è¡Œæ‰“å‡» (Strike)ï¼Œåˆ†é…äº† ${dmg} ä¼¤å®³ï¼<br>`;
        wizardState.meleeLogs += msg;
        
        targetOpponent.applyWounds(dmg);
        playSound('heavy_strike');
        triggerCombatVisual("âš”ï¸ STRIKE! -" + dmg, "strike");
      } else {
        let targetIdx = -1;
        if (dice.isCrit) {
          targetIdx = opponentDiceList.findIndex(d => !d.used && d.isCrit);
          if (targetIdx === -1) {
            targetIdx = opponentDiceList.findIndex(d => !d.used);
          }
        } else {
          targetIdx = opponentDiceList.findIndex(d => !d.used && !d.isCrit);
        }

        if (targetIdx === -1) {
          playSound('alert');
          alert('æ²¡æœ‰åˆæ³•çš„å¯¹æ–¹éª°å­å¯ä¾›æ ¼æŒ¡æ‹›æž¶ï¼');
          return;
        }

        dice.used = true;
        opponentDiceList[targetIdx].used = true;
        const msg = `> ${side === 'attacker' ? 'æ”»å‡»æ–¹' : 'é˜²å®ˆæ–¹'} æ‰§è¡Œæ ¼æŒ¡ (Parry)ï¼Œæ¶ˆåŽ»å¯¹æ–¹ä¸€ä¸ªéª°å­ [${opponentDiceList[targetIdx].val}]ï¼<br>`;
        wizardState.meleeLogs += msg;
        playSound('metal_clash');
        triggerCombatVisual("ðŸ›¡ï¸ PARRY!", "parry");
      }

      const opponentSide = side === 'attacker' ? 'defender' : 'attacker';
      const oppWounds = opponentSide === 'attacker' ? wizardState.attacker.wounds : wizardState.defender.wounds;
      const oppDiceList = opponentSide === 'attacker' ? wizardState.activeAttackerDice : wizardState.activeDefenderDice;
      const oppHasUsable = oppDiceList.some(d => !d.used) && oppWounds > 0;
      
      const ownWounds = side === 'attacker' ? wizardState.attacker.wounds : wizardState.defender.wounds;
      const ownDiceList = side === 'attacker' ? wizardState.activeAttackerDice : wizardState.activeDefenderDice;
      const ownHasUsable = ownDiceList.some(d => !d.used) && ownWounds > 0;

      if (oppHasUsable && ownHasUsable) {
        wizardState.meleeTurn = opponentSide;
      } else if (oppHasUsable) {
        wizardState.meleeTurn = opponentSide;
      } else if (ownHasUsable) {
        wizardState.meleeTurn = side;
      }

      wizardState.selectedMeleeDice = null;
      renderFightStep();
      
      if (action === 'strike') {
        triggerAvatarHitEffect(targetOpponent.id, 'melee');
      }
    }

    function rollDicePool(count) {
      const rolls = [];
      for (let i = 0; i < count; i++) {
        rolls.push(Math.floor(Math.random() * 6) + 1);
      }
      return rolls.sort((a, b) => b - a);
    }

    function evaluateAttackRolls(rolls, bs) {
      let criticals = 0;
      let normals = 0;
      let misses = 0;
      for (const val of rolls) {
        if (val === 6) {
          criticals++;
        } else if (val >= bs) {
          normals++;
        } else {
          misses++;
        }
      }
      return { criticals, normals, misses, rolls };
    }

    function evaluateDefenseRolls(rolls, sv) {
      let criticals = 0;
      let normals = 0;
      let fails = 0;
      for (const val of rolls) {
        if (val === 6) {
          criticals++;
        } else if (val >= sv) {
          normals++;
        } else {
          fails++;
        }
      }
      return { criticals, normals, fails, rolls };
    }
